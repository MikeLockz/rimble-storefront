# Rimble UI

## Installation

```lang=jsx
    npm install --save rimble-ui styled-components
```
## Usage
```lang=jsx
    import React, { Component } from 'react'

    import { Button } from 'rimble-ui'

    class Example extends Component {
      render () {
        return (
          <Button size={'medium'}>
            Click me!
          </Button>
        )
      }
    }
    ```
